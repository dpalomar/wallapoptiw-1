25  	public static boolean validarMail(String mail) {
26  		Pattern plantilla = Pattern.compile("^[A-Za-z0-9._%-]+@[A-Za-z0-9.-]+\\.[a-zA-Z]{2,4}$");
27  		Matcher resultado = plantilla.matcher(mail);
28  		if(resultado.find()==true){
29  			return true;
30  		}else return false;
31  	}
